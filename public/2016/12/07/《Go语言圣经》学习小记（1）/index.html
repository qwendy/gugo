func forwardRotate(s []int, x int) {
	mid := x
	k := mid
	for i := 0; i < len(s); i++ {
		s[i], s[k] = s[k], s[i]
		k++
		if i == mid {
			if k == len(s) {
				return
			}
			mid = k
		} else if k == len(s) {
			k = mid
		}
	}
}

func noRepeatSide(s []string) []string {
	count := len(s)
	for i := range s {
		if i == len(s)-1 {
			return s
		}
		if s[i] == s[i+1] {
			s = append(s[:i], s[i+1:]...)
			count--
		}
	}
	return s
}
